# View current disks
lsblk
sudo fdisk -l

# Start partitioning a new disk
sudo fdisk /dev/sdb

# Inside fdisk, use:
n  # New partition
p  # Primary
w  # Write changes

# Format and mount
sudo mkfs.ext4 /dev/sdb1
sudo mkdir /mnt/mydata
sudo mount /dev/sdb1 /mnt/mydata

# Permanent mount
sudo nano /etc/fstab
# Add line:
# /dev/sdb1 /mnt/mydata ext4 defaults 0 0

# Create LVM-compatible partition
sudo fdisk /dev/sdb   â†’ type 8e for LVM

# Initialize Physical Volume
sudo pvcreate /dev/sdb1

# Create Volume Group
sudo vgcreate myvg /dev/sdb1

# Create Logical Volume
sudo lvcreate -L 1G -n mylv myvg

# Format and mount
sudo mkfs.ext4 /dev/myvg/mylv
sudo mkdir /mnt/lvmdata
sudo mount /dev/myvg/mylv /mnt/lvmdata

# Add to fstab for auto-mount
/dev/myvg/mylv /mnt/lvmdata ext4 defaults 0 0
